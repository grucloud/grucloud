Table

Manages an DynamoDB Table.

Sample code​

exports.createResources = () => [
  {
    type: "Table",
    group: "DynamoDB",
    name: "myTable",
    properties: ({}) => ({
      AttributeDefinitions: [
        {
          AttributeName: "Id",
          AttributeType: "S",
        },
      ],
      KeySchema: [
        {
          AttributeName: "Id",
          KeyType: "HASH",
        },
      ],
      ProvisionedThroughput: {
        ReadCapacityUnits: 5,
        WriteCapacityUnits: 5,
      },
      Tags: [
        {
          Key: "TOTOKEY",
          Value: "TOTO",
        },
      ],
    }),
  },
];
