File System

Provides an EFS File System.

Examples​

exports.createResources = () => [
  {
    type: "FileSystem",
    group: "EFS",
    name: "fs-0aaaf7b0715648e5b",
    properties: ({}) => ({
      Encrypted: false,
      PerformanceMode: "generalPurpose",
      ThroughputMode: "bursting",
    }),
  },
];
